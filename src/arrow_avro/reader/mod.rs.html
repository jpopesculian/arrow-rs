18//! Avro reader
19//!
20//! This module provides facilities to read Apache Avro-encoded files or streams
21//! into Arrow's `RecordBatch` format. In particular, it introduces:
22//!
23//! * `ReaderBuilder`: Configures Avro reading, e.g., batch size
24//! * `Reader`: Yields `RecordBatch` values, implementing `Iterator`
25//! * `Decoder`: A low-level push-based decoder for Avro records
26//!
27//! # Basic Usage
28//!
29//! `Reader` can be used directly with synchronous data sources, such as [`std::fs::File`].
30//!
31//! ## Reading a Single Batch
32//!
33//! ```
34//! # use std::fs::File;
35//! # use std::io::BufReader;
36//! # use arrow_avro::reader::ReaderBuilder;
37//!
38//! let file = File::open("../testing/data/avro/alltypes_plain.avro").unwrap();
39//! let mut avro = ReaderBuilder::new().build(BufReader::new(file)).unwrap();
40//! let batch = avro.next().unwrap();
41//! ```
42//!
43//! # Async Usage
44//!
45//! The lower-level `Decoder` can be integrated with various forms of async data streams,
46//! and is designed to be agnostic to different async IO primitives within
47//! the Rust ecosystem. It works by incrementally decoding Avro data from byte slices.
48//!
49//! For example, see below for how it could be used with an arbitrary `Stream` of `Bytes`:
50//!
51//! ```
52//! # use std::task::{Poll, ready};
53//! # use bytes::{Buf, Bytes};
54//! # use arrow_schema::ArrowError;
55//! # use futures::stream::{Stream, StreamExt};
56//! # use arrow_array::RecordBatch;
57//! # use arrow_avro::reader::Decoder;
58//!
59//! fn decode_stream<S: Stream<Item = Bytes> + Unpin>(
60//!     mut decoder: Decoder,
61//!     mut input: S,
62//! ) -> impl Stream<Item = Result<RecordBatch, ArrowError>> {
63//!     let mut buffered = Bytes::new();
64//!     futures::stream::poll_fn(move |cx| {
65//!         loop {
66//!             if buffered.is_empty() {
67//!                 buffered = match ready!(input.poll_next_unpin(cx)) {
68//!                     Some(b) => b,
69//!                     None => break,
70//!                 };
71//!             }
72//!             let decoded = match decoder.decode(buffered.as_ref()) {
73//!                 Ok(decoded) => decoded,
74//!                 Err(e) => return Poll::Ready(Some(Err(e))),
75//!             };
76//!             let read = buffered.len();
77//!             buffered.advance(decoded);
78//!             if decoded != read {
79//!                 break
80//!             }
81//!         }
82//!         // Convert any fully-decoded rows to a RecordBatch, if available
83//!         Poll::Ready(decoder.flush().transpose())
84//!     })
85//! }
86//! ```
87//!

104/// Read the Avro file header (magic, metadata, sync marker) from `reader`.
105fn read_header<R: BufRead>(mut reader: R) -> Result<Header, ArrowError> {
106    let mut decoder = HeaderDecoder::default();
107    loop {
108        let buf = reader.fill_buf()?;
109        if buf.is_empty() {
110            break;
111        }
112        let read = buf.len();
113        let decoded = decoder.decode(buf)?;
114        reader.consume(decoded);
115        if decoded != read {
116            break;
117        }
118    }
119    decoder.flush().ok_or_else(|| {
120        ArrowError::ParseError("Unexpected EOF while reading Avro header".to_string())
121    })
122}

124/// A low-level interface for decoding Avro-encoded bytes into Arrow `RecordBatch`.
125#[derive(Debug)]
126pub struct Decoder {
127    record_decoder: RecordDecoder,
128    batch_size: usize,
129    decoded_rows: usize,
130}
131
132impl Decoder {
133    fn new(record_decoder: RecordDecoder, batch_size: usize) -> Self {
134        Self {
135            record_decoder,
136            batch_size,
137            decoded_rows: 0,
138        }
139    }
140
141    /// Return the Arrow schema for the rows decoded by this decoder
142    pub fn schema(&self) -> SchemaRef {
143        self.record_decoder.schema().clone()
144    }
145
146    /// Return the configured maximum number of rows per batch
147    pub fn batch_size(&self) -> usize {
148        self.batch_size
149    }
150
151    /// Feed `data` into the decoder row by row until we either:
152    /// - consume all bytes in `data`, or
153    /// - reach `batch_size` decoded rows.
154    ///
155    /// Returns the number of bytes consumed.
156    pub fn decode(&mut self, data: &[u8]) -> Result<usize, ArrowError> {
157        let mut total_consumed = 0usize;
158        while total_consumed < data.len() && self.decoded_rows < self.batch_size {
159            let consumed = self.record_decoder.decode(&data[total_consumed..], 1)?;
160            if consumed == 0 {
161                break;
162            }
163            total_consumed += consumed;
164            self.decoded_rows += 1;
165        }
166        Ok(total_consumed)
167    }
168
169    /// Produce a `RecordBatch` if at least one row is fully decoded, returning
170    /// `Ok(None)` if no new rows are available.
171    pub fn flush(&mut self) -> Result<Option<RecordBatch>, ArrowError> {
172        if self.decoded_rows == 0 {
173            Ok(None)
174        } else {
175            let batch = self.record_decoder.flush()?;
176            self.decoded_rows = 0;
177            Ok(Some(batch))
178        }
179    }
180
181    /// Returns the number of rows that can be added to this decoder before it is full.
182    pub fn capacity(&self) -> usize {
183        self.batch_size.saturating_sub(self.decoded_rows)
184    }
185
186    /// Returns true if the decoder has reached its capacity for the current batch.
187    pub fn batch_is_full(&self) -> bool {
188        self.capacity() == 0
189    }
190}

192/// A builder to create an [`Avro Reader`](Reader) that reads Avro data
193/// into Arrow `RecordBatch`.
194#[derive(Debug)]
195pub struct ReaderBuilder {
196    batch_size: usize,
197    strict_mode: bool,
198    utf8_view: bool,
199    schema: Option<AvroSchema<'static>>,
200}
201
202impl Default for ReaderBuilder {
203    fn default() -> Self {
204        Self {
205            batch_size: 1024,
206            strict_mode: false,
207            utf8_view: false,
208            schema: None,
209        }
210    }
211}
212
213impl ReaderBuilder {
214    /// Creates a new [`ReaderBuilder`] with default settings:
215    /// - `batch_size` = 1024
216    /// - `strict_mode` = false
217    /// - `utf8_view` = false
218    /// - `schema` = None
219    pub fn new() -> Self {
220        Self::default()
221    }
222
223    fn make_record_decoder(&self, schema: &AvroSchema<'_>) -> Result<RecordDecoder, ArrowError> {
224        let root_field = AvroField::try_from(schema)?;
225        RecordDecoder::try_new_with_options(
226            root_field.data_type(),
227            self.utf8_view,
228            self.strict_mode,
229        )
230    }
231
232    fn build_impl<R: BufRead>(self, reader: &mut R) -> Result<(Header, Decoder), ArrowError> {
233        let header = read_header(reader)?;
234        let record_decoder = if let Some(schema) = &self.schema {
235            self.make_record_decoder(schema)?
236        } else {
237            let avro_schema: Option<AvroSchema<'_>> = header
238                .schema()
239                .map_err(|e| ArrowError::ExternalError(Box::new(e)))?;
240            let avro_schema = avro_schema.ok_or_else(|| {
241                ArrowError::ParseError("No Avro schema present in file header".to_string())
242            })?;
243            self.make_record_decoder(&avro_schema)?
244        };
245        let decoder = Decoder::new(record_decoder, self.batch_size);
246        Ok((header, decoder))
247    }
248
249    /// Sets the row-based batch size
250    pub fn with_batch_size(mut self, batch_size: usize) -> Self {
251        self.batch_size = batch_size;
252        self
253    }
254
255    /// Set whether to use StringViewArray for string data
256    ///
257    /// When enabled, string data from Avro files will be loaded into
258    /// Arrow's StringViewArray instead of the standard StringArray.
259    pub fn with_utf8_view(mut self, utf8_view: bool) -> Self {
260        self.utf8_view = utf8_view;
261        self
262    }
263
264    /// Get whether StringViewArray is enabled for string data
265    pub fn use_utf8view(&self) -> bool {
266        self.utf8_view
267    }
268
269    /// Controls whether certain Avro unions of the form `[T, "null"]` should produce an error.
270    pub fn with_strict_mode(mut self, strict_mode: bool) -> Self {
271        self.strict_mode = strict_mode;
272        self
273    }
274
275    /// Sets the Avro schema.
276    ///
277    /// If a schema is not provided, the schema will be read from the Avro file header.
278    pub fn with_schema(mut self, schema: AvroSchema<'static>) -> Self {
279        self.schema = Some(schema);
280        self
281    }

283    /// Create a [`Reader`] from this builder and a `BufRead`
284    pub fn build<R: BufRead>(self, mut reader: R) -> Result<Reader<R>, ArrowError> {
285        let (header, decoder) = self.build_impl(&mut reader)?;
286        Ok(Reader {
287            reader,
288            header,
289            decoder,
290            block_decoder: BlockDecoder::default(),
291            block_data: Vec::new(),
292            block_cursor: 0,
293            finished: false,
294        })
295    }
296
297    /// Create a [`Decoder`] from this builder and a `BufRead` by
298    /// reading and parsing the Avro file's header. This will
299    /// not create a full [`Reader`].
300    pub fn build_decoder<R: BufRead>(self, mut reader: R) -> Result<Decoder, ArrowError> {
301        match self.schema {
302            Some(ref schema) => {
303                let record_decoder = self.make_record_decoder(schema)?;
304                Ok(Decoder::new(record_decoder, self.batch_size))
305            }
306            None => {
307                let (_, decoder) = self.build_impl(&mut reader)?;
308                Ok(decoder)
309            }
310        }
311    }
312}

314/// A high-level Avro `Reader` that reads container-file blocks
315/// and feeds them into a row-level [`Decoder`].
316#[derive(Debug)]
317pub struct Reader<R: BufRead> {
318    reader: R,
319    header: Header,
320    decoder: Decoder,
321    block_decoder: BlockDecoder,
322    block_data: Vec<u8>,
323    block_cursor: usize,
324    finished: bool,
325}
326
327impl<R: BufRead> Reader<R> {
328    /// Return the Arrow schema discovered from the Avro file header
329    pub fn schema(&self) -> SchemaRef {
330        self.decoder.schema()
331    }
332
333    /// Return the Avro container-file header
334    pub fn avro_header(&self) -> &Header {
335        &self.header
336    }
337
338    /// Reads the next [`RecordBatch`] from the Avro file or `Ok(None)` on EOF
339    fn read(&mut self) -> Result<Option<RecordBatch>, ArrowError> {
340        'outer: while !self.finished && !self.decoder.batch_is_full() {
341            while self.block_cursor == self.block_data.len() {
342                let buf = self.reader.fill_buf()?;
343                if buf.is_empty() {
344                    self.finished = true;
345                    break 'outer;
346                }
347                // Try to decode another block from the buffered reader.
348                let consumed = self.block_decoder.decode(buf)?;
349                self.reader.consume(consumed);
350                if let Some(block) = self.block_decoder.flush() {
351                    // Successfully decoded a block.
352                    let block_data = if let Some(ref codec) = self.header.compression()? {
353                        codec.decompress(&block.data)?
354                    } else {
355                        block.data
356                    };
357                    self.block_data = block_data;
358                    self.block_cursor = 0;
359                } else if consumed == 0 {
360                    // The block decoder made no progress on a non-empty buffer.
361                    return Err(ArrowError::ParseError(
362                        "Could not decode next Avro block from partial data".to_string(),
363                    ));
364                }
365            }
366            // Try to decode more rows from the current block.
367            let consumed = self.decoder.decode(&self.block_data[self.block_cursor..])?;
368            if consumed == 0 && self.block_cursor < self.block_data.len() {
369                self.block_cursor = self.block_data.len();
370            } else {
371                self.block_cursor += consumed;
372            }
373        }
374        self.decoder.flush()
375    }
376}

378impl<R: BufRead> Iterator for Reader<R> {
379    type Item = Result<RecordBatch, ArrowError>;
380
381    fn next(&mut self) -> Option<Self::Item> {
382        self.read().transpose()
383    }
384}
385
386impl<R: BufRead> RecordBatchReader for Reader<R> {
387    fn schema(&self) -> SchemaRef {
388        self.schema()
389    }
390}

413    fn read_file(path: &str, batch_size: usize, utf8_view: bool) -> RecordBatch {
414        let file = File::open(path).unwrap();
415        let reader = ReaderBuilder::new()
416            .with_batch_size(batch_size)
417            .with_utf8_view(utf8_view)
418            .build(BufReader::new(file))
419            .unwrap();
420        let schema = reader.schema();
421        let batches = reader.collect::<Result<Vec<_>, _>>().unwrap();
422        arrow::compute::concat_batches(&schema, &batches).unwrap()
423    }
424
425    fn decode_stream<S: Stream<Item = Bytes> + Unpin>(
426        mut decoder: Decoder,
427        mut input: S,
428    ) -> impl Stream<Item = Result<RecordBatch, ArrowError>> {
429        async_stream::try_stream! {
430            if let Some(data) = input.next().await {
431                let consumed = decoder.decode(&data)?;
432                if consumed < data.len() {
433                    Err(ArrowError::ParseError(
434                        "did not consume all bytes".to_string(),
435                    ))?;
436                }
437            }
438            if let Some(batch) = decoder.flush()? {
439                yield batch
440            }
441        }
442    }

587    fn test_decode_stream_with_schema() {
588        struct TestCase<'a> {
589            name: &'a str,
590            schema: &'a str,
591            expected_error: Option<&'a str>,
592        }
593        let tests = vec![
594            TestCase {
595                name: "success",
596                schema: r#"{"type":"record","name":"test","fields":[{"name":"f2","type":"string"}]}"#,
597                expected_error: None,
598            },
599            TestCase {
600                name: "valid schema invalid data",
601                schema: r#"{"type":"record","name":"test","fields":[{"name":"f2","type":"long"}]}"#,
602                expected_error: Some("did not consume all bytes"),
603            },
604        ];
605        for test in tests {
606            let schema_s2: crate::schema::Schema = serde_json::from_str(test.schema).unwrap();
607            let record_val = "some_string";
608            let mut body = vec![];
609            body.push((record_val.len() as u8) << 1);
610            body.extend_from_slice(record_val.as_bytes());
611            let mut reader_placeholder = Cursor::new(&[] as &[u8]);
612            let builder = ReaderBuilder::new()
613                .with_batch_size(1)
614                .with_schema(schema_s2);
615            let decoder_result = builder.build_decoder(&mut reader_placeholder);
616            let decoder = match decoder_result {
617                Ok(decoder) => decoder,
618                Err(e) => {
619                    if let Some(expected) = test.expected_error {
620                        assert!(
621                            e.to_string().contains(expected),
622                            "Test '{}' failed: unexpected error message at build.\nExpected to contain: '{expected}'\nActual: '{e}'",
623                            test.name,
624                        );
625                        continue;
626                    } else {
627                        panic!("Test '{}' failed at decoder build: {e}", test.name);
628                    }
629                }
630            };
631            let stream = Box::pin(stream::once(async { Bytes::from(body) }));
632            let decoded_stream = decode_stream(decoder, stream);
633            let batches_result: Result<Vec<RecordBatch>, ArrowError> =
634                block_on(decoded_stream.try_collect());
635            match (batches_result, test.expected_error) {
636                (Ok(batches), None) => {
637                    let batch =
638                        arrow::compute::concat_batches(&batches[0].schema(), &batches).unwrap();
639                    let expected_field = Field::new("f2", DataType::Utf8, false);
640                    let expected_schema = Arc::new(Schema::new(vec![expected_field]));
641                    let expected_array = Arc::new(StringArray::from(vec![record_val]));
642                    let expected_batch =
643                        RecordBatch::try_new(expected_schema, vec![expected_array]).unwrap();
644                    assert_eq!(batch, expected_batch, "Test '{}' failed", test.name);
645                    assert_eq!(
646                        batch.schema().field(0).name(),
647                        "f2",
648                        "Test '{}' failed",
649                        test.name
650                    );
651                }
652                (Err(e), Some(expected)) => {
653                    assert!(
654                        e.to_string().contains(expected),
655                        "Test '{}' failed: unexpected error message at decode.\nExpected to contain: '{expected}'\nActual: '{e}'",
656                        test.name,
657                    );
658                }
659                (Ok(batches), Some(expected)) => {
660                    panic!(
661                        "Test '{}' was expected to fail with '{expected}', but it succeeded with: {:?}",
662                        test.name, batches
663                    );
664                }
665                (Err(e), None) => {
666                    panic!(
667                        "Test '{}' was not expected to fail, but it did with '{e}'",
668                        test.name
669                    );
670                }
671            }
672        }
673    }
